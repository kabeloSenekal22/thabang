import java.util.Scanner;
public class AgeCalculation
{

	{

		//Declare variables
		int iAge, iCurrentYear, iBirthYear;
		
		//Inputs
		System.out.print("Enter your current birth year :");
		iBirthYear = sc.nextInt();
		System.out.print("Enter the current :");
